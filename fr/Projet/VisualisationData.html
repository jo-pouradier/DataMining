## number of element per value of each columns
plt.figure(figsize=(25,17))
# set title
plt.suptitle('Number of element per value of each columns', fontsize=30)
# add space between subplots
plt.subplots_adjust(hspace=0.3, wspace=0.1)

# loop over columns
for col in df.columns:
    plt.subplot(4,3,df.columns.get_loc(col)+1)
    df[col].value_counts().plot(kind='bar', title=col, color=['red', 'green', 'blue','purple'], rot=45)
