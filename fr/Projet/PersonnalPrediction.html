import random

def randomImage():
    dir1 = "images"
    themes = ["Exoplanet", "Moto", "Pokemon2", "Voitures"]
    dir2 = random.choice(themes)
    dir = os.path.join(dir1, dir2)
    files = os.listdir(dir)
    try:
        files.remove(".DS_Store")
        # remove all csv
        files = [f for f in files if f[-3:] != "csv"]
    except:
        pass
    file = random.choice(files)
    return file
