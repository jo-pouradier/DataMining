# getting all images
import os
from PIL import Image
import numpy as np
themes = ["Exoplanet", "Moto","Pokemon2", "Voitures"]
width_possible = [120, 640, 960, 1280, 1600, 1920, 2880]
height_possible = [x*9/16 for x in width_possible]
height_possible[0] = 120

#making correspond width and height
sizes = list(zip(width_possible, height_possible))
print(sizes)
sizes = np.array(sizes)

[(120, 120), (640, 360.0), (960, 540.0), (1280, 720.0), (1600, 900.0), (1920, 1080.0), (2880, 1620.0)]

def get_new_size(img_path):
    img = Image.open(img_path)
    size = img.size
    print(size)
    size = np.array(size)
    # get the closest height np.abs(np.array(height_possible) - height).argmin()
    sizes_distance = np.sqrt(np.sum((sizes-size)**2, axis=1))
    index_new_size = np.where(sizes_distance == np.amin(sizes_distance))
    new_size = sizes[index_new_size]
    print(new_size[0])
    # resize image
    img = img.resize((int(new_size[0][0]), int(new_size[0][1])))
    # save image
    img.mode = "RGB"
    img.save(img_path)
