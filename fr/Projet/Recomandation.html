print(data.head())

# label encoder
themeLabel = LabelEncoder()

heightLabel = LabelEncoder()
formatLabel = LabelEncoder()
orientationLabel = LabelEncoder()
color_0Label = LabelEncoder()
color_1Label = LabelEncoder()
color_2Label = LabelEncoder()
color_3Label = LabelEncoder()
color_4Label = LabelEncoder()
redLabel = LabelEncoder()
greenLabel = LabelEncoder()
blueLabel = LabelEncoder()

labels = [themeLabel, widthLabel, heightLabel, formatLabel, orientationLabel, color_0Label, color_1Label, color_2Label, color_3Label, color_4Label, redLabel, greenLabel, blueLabel]
columns = ['theme', 'width', 'height', 'format', 'orientation', 'color_0', 'color_1', 'color_2', 'color_3', 'color_4', 'red', 'green', 'blue']

data = data.drop('image', axis=1, inplace=True)

for i in range(len(labels)):
    data[columns[i]] = labels[i].fit_transform(data[columns[i]])
